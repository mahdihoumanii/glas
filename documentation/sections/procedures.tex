\section{FORM Procedure Library}
\label{sec:procedures}
%=============================================================================

\textsc{GLAS} includes a library of \textsc{FORM} procedures in \texttt{resources/formlib/procedures/}. The key procedures are:

\subsection*{Physics Rules and Algebra}
\begin{itemize}
    \item \texttt{FeynmanRules.prc}: Applies QCD Feynman rules, converting QGRAF fields to spinors and polarization vectors, inserting vertices and propagators, and normalizing Lorentz and color structures.
    \item \texttt{MassCT.prc}: Implements mass counterterm insertions for massive fermions, including the explicit $\delta m$ term and bookkeeping for the coupling power.
    \item \texttt{DiracSimplify.prc}: Performs gamma-matrix algebra, contracts Lorentz indices, applies on-shell relations, and reduces spinor chains to trace form.
    \item \texttt{Conjugate.prc}: Complex-conjugates amplitudes, reverses spinor chains, and swaps polarization vectors and color structures consistently.
    \item \texttt{PolarizationSums.prc}: Executes polarization sums and traces for external states, producing contracted chains ready for scalar integral extraction.
    \item \texttt{color.prc}: Reduces SU($N$) color structures, expands traces, contracts adjoint indices, and substitutes SU(3) constants.
\end{itemize}

\subsection*{Kinematic Substitutions}
\begin{itemize}
    \item \texttt{mandelstam2x2.prc}: Rewrites dot products for $2\to2$ kinematics in terms of Mandelstam variables and masses.
    \item \texttt{mandelstam2x3.prc}: Rewrites dot products for $2\to3$ kinematics in terms of Mandelstam variables and masses.
\end{itemize}

\subsection*{Rational Function Handling}
\begin{itemize}
    \item \texttt{SymToRat.prc}: Converts kinematic symbols and denominators into rational-function placeholders for downstream collection.
    \item \texttt{RationalFunction.prc}: Simplifies \texttt{den(x)*x = 1} if x is a combination of terms (e.g. $s_{12}*den(s_{12}-m_t^2) - m_t^2*den(s_{12}-m_t^2) = 1$). And this is done for each denominator alone.
    \item \texttt{PolyRat.prc}: Promotes selected symbols to rational form and expands \texttt{rat} objects into \texttt{den}-based expressions.
    \item \texttt{Together.prc}: Takes common denominator and keeps the output in terms of the \texttt{FORM PolyRatFun rat(num,den)}.
    \item \texttt{rationals.prc}: Groups identical denominators, builds polynomial rational functions like \eqref{den_group} before applying \texttt{linrels}.
    \item \texttt{toden.prc}: Rewrites inverse kinematic invariants into explicit denominator factors.
\end{itemize}
